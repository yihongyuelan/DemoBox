Classification and Encapsulation
In the code, if you need to return different objects depending on the type, or different operations, you can use a simple factory pattern. Extract the relevant code and encapsulated into a method, depending on the type of return or be treated differently.

The key word for Simple Factory are:Encapsulation and Decoupling.

The following is an example of the production of Pizza. 

/**
 * Order pizza process
 */
Pizza orderPizza(String type) {
		Pizza pizza = null;
		if (type.equals("cheese")) {
			pizza = new CheesePizza();
		} else if (type.equals("pepperoni")) {
			pizza = new PepperoniPizza();
		} else if (type.equals("clam")) {
			pizza = new ClamPizza();
		}else if (type.equals("veggie")) {
			pizza = new VeggiePizza();
		}
		pizza.prpare();
		pizza.bake();
		pizza.cut();
		pizza.box();
		return pizza;
}
				

/**
 * Encapsulating object creation
 */
public class PizzaStore {
	SimplePizzaFactory factory;

	public PizzaStore(SimplePizzaFactory factory) {
		this.factory = factory;
	}

	Pizza orderPizza(String type) {
		Pizza pizza = null;
		pizza = factory.createPizza(type);
		pizza.prpare();
		pizza.bake();
		pizza.cut();
		pizza.box();
		return pizza;
	}
}
/**
 * Building a simple pizza factory
 */
public class SimplePizzaFactory {

	public Pizza createPizza(String type) {
		Pizza pizza = null;
		if (type.equals("cheese")) {
			pizza = new CheesePizza();
		} else if (type.equals("pepperoni")) {
			pizza = new PepperoniPizza();
		} else if (type.equals("clam")) {
			pizza = new ClamPizza();
		}else if (type.equals("veggie")) {
			pizza = new VeggiePizza();
		}
		return pizza;
	}
}
